        public void ContarAnas(List<Empleados> empleados)
        {
            var cont = empleados.Count(nombre => nombre.Nombres.ToUpper().Equals("ANA"));

            Console.WriteLine("Hay {0} trabajadores en la lista que se llama Ana", cont);
        }

                            select emp.Salario).DefaultIfEmpty(0).Average();
